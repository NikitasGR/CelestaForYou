    <button class="btn" id="btn2" onclick="answer(1)">Απάντηση 2</button>
    <button class="btn" id="btn3" onclick="answer(2)">Απάντηση 3</button>
    <button class="btn" id="btn4" onclick="answer(3)">Απάντηση 4</button>

<script>
const questions = [
  {
    question: "Σε ποια σειρά ακούγεται η φράση: 'Τι έγινε ρε παιδιά;'",
    answers: ["Εγκλήματα", "Στο Παρά Πέντε", "Κωνσταντίνου και Ελένης", "Οι Απαράδεκτοι"],
    correct: 3
  },
  {
    question: "Πώς λέγεται η μάνα της Μαρίνας στο 'Ευτυχισμένοι Μαζί';",
    answers: ["Πόπη", "Βίκυ", "Μαίρη", "Νίκη"],
    correct: 0
  },
  {
    question: "Ποιος είναι ο καλύτερος φίλος του Χάρη στο 'Κωνσταντίνου και Ελένης';",
    answers: ["Νικόλας", "Μάνθος", "Τάσος", "Μιχάλης"],
    correct: 1
  },
  {
    question: "Ποια σειρά έχει χαρακτήρα τον Θανάση Βούτσα ως παππού;",
    answers: ["Το Καφέ της Χαράς", "Μην Αρχίζεις τη Μουρμούρα", "Ευτυχισμένοι Μαζί", "Λατρεμένοι μου Γείτονες"],
    correct: 2
  }
];

let currentQuestion = 0;
let score = 0;

function loadQuestion() {
  const q = questions[currentQuestion];
  document.getElementById('question').innerText = q.question;
  document.getElementById('btn1').innerText = q.answers[0];
  document.getElementById('btn2').innerText = q.answers[1];
  document.getElementById('btn3').innerText = q.answers[2];
  document.getElementById('btn4').innerText = q.answers[3];
  document.getElementById('score').innerText = `Σκορ: ${score}/${questions.length}`;
}

function answer(index) {
  if (index === questions[currentQuestion].correct) {
    score++;
    alert("Σωστό! 🎉");
  } else {
    alert("Λάθος! 😢");
  }
  currentQuestion++;
  if (currentQuestion < questions.length) {
    loadQuestion();
  } else {
    document.getElementById('question').innerText = "Τέλος Παιχνιδιού!";
    document.getElementById('btn1').style.display = 'none';
    document.getElementById('btn2').style.display = 'none';
    document.getElementById('btn3').style.display = 'none';
    document.getElementById('btn4').style.display = 'none';
    document.getElementById('score').innerText = `Τελικό Σκορ: ${score}/${questions.length}`;
  }
}

function restart() {
  currentQuestion = 0;
  score = 0;
  document.getElementById('btn1').style.display = 'block';
  document.getElementById('btn2').style.display = 'block';
  document.getElementById('btn3').style.display = 'block';
  document.getElementById('btn4').style.display = 'block';
  loadQuestion();
}

window.onload = loadQuestion;
</script>
